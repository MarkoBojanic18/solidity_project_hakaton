pragma solidity ^0.8.4;

contract Doctor{

    address private doctorAccount;  //address of doctor
    string private first_name;   //first name of doctor
    string private last_name;    //last name of doctor
    string private role;        //position of the doctor

    constructor(address _doctorAccount,string memory _first_name,string memory _last_name,string memory _role){
        doctorAccount = _doctorAccount;
        first_name = _first_name;
        last_name = _last_name;
        role = _role;
    }


    function getDoctorAccount() public view returns(address){
        return doctorAccount;
    }

    function setDoctorAccount(address _doctorAccount) public{
        doctorAccount = _doctorAccount;
    }

    function getDoctorFirstName() public view returns(string memory){
        return first_name;
    }

    function setDoctorFirstName(string memory _first_name) public{
        first_name = _first_name;
    }

    function getDoctorLastName() public view returns(string memory){
        return last_name;
    }

    function setDoctorLastName(string memory _last_name) public{
        last_name = _last_name;
    }

    function getDoctorRole() public view returns(string memory){
        return role;
    }

    function setDoctorRole(string memory _role) public{
        role = _role;
    }


}
